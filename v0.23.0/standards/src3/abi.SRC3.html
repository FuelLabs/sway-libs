Abi standards::src3::SRC3

abi SRC3 {
    /// Mints new assets using the `sub_id` sub-identifier.
    ///
    /// # Arguments
    ///
    /// * `recipient`: [Identity] - The user to which the newly minted asset is transferred to.
    /// * `sub_id`: [SubId] - The sub-identifier of the newly minted asset.
    /// * `amount`: [u64] - The quantity of coins to mint.
    ///
    /// # Examples
    ///
    /// ```sway
    /// use src3::SRC3;
    ///
    /// fn foo(contract_id: ContractId) {
    ///     let contract_abi = abi(SR3, contract);
    ///     contract_abi.mint(Identity::ContractId(contract_id), SubId::zero(), 100);
    /// }
    /// ```
    #[storage(read, write)]
    fn mint(recipient: Identity, sub_id: SubId, amount: u64);
    /// Burns assets sent with the given `sub_id`.
    ///
    /// # Additional Information
    ///
    /// NOTE: The sha-256 hash of `(ContractId, SubId)` must match the `AssetId` where `ContractId` is the id of
    /// the implementing contract and `SubId` is the given `sub_id` argument.
    ///
    /// # Arguments
    ///
    /// * `sub_id`: [SubId] - The sub-identifier of the asset to burn.
    /// * `amount`: [u64] - The quantity of coins to burn.
    ///
    /// # Examples
    ///
    /// ```sway
    /// use src3::SRC3;
    ///
    /// fn foo(contract_id: ContractId, asset_id: AssetId) {
    ///     let contract_abi = abi(SR3, contract_id);
    ///     contract_abi {
    ///         gas: 10000,
    ///         coins: 100,
    ///         asset_id: asset_id,
    ///     }.burn(SubId::zero(), 100);
    /// }
    /// ```
    #[payable]
    #[storage(read, write)]
    fn burn(sub_id: SubId, amount: u64);
}

fn burn(
sub_id: SubId,
amount: u64,
)

Burns assets sent with the given sub_id.

Additional Information

NOTE: The sha-256 hash of (ContractId, SubId) must match the AssetId where ContractId is the id of
the implementing contract and SubId is the given sub_id argument.

Arguments

  • sub_id: [SubId] - The sub-identifier of the asset to burn.
  • amount: [u64] - The quantity of coins to burn.

Examples

use src3::SRC3;

fn foo(contract_id: ContractId, asset_id: AssetId) {
    let contract_abi = abi(SR3, contract_id);
    contract_abi {
        gas: 10000,
        coins: 100,
        asset_id: asset_id,
    }.burn(SubId::zero(), 100);
}
